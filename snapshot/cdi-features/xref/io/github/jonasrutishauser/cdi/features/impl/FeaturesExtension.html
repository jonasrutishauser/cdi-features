86      void discoverFeatures(@Observes ProcessBean<?> bean, BeanManager beanManager) {
87          Optional<Feature> feature = feature(bean.getBean());
88          if (feature.isPresent()) {
89              validate(feature.get(), bean.getBean(), beanManager, bean::addDefinitionError);
90              featureBeans.add(bean.getBean());
91              if (useInterceptor) {
92                  invokerFactory.processFeatureBean(bean);
93              }
94          }
95      }

133     private void validate(Feature feature, Bean<?> bean, BeanManager beanManager, Consumer<Throwable> definitionError) {
134         if (feature.remaining()) {
135             if (hasDefinedSelector(feature)) {
136                 definitionError.accept(
137                         new IllegalStateException("selector must not be set if remaining is true on bean " + bean));
138             }
139             if (isDefined(feature.propertyKey())) {
140                 definitionError.accept(
141                         new IllegalStateException("propertyKey must not be set if remaining is true on bean " + bean));
142             }
143         } else if (hasDefinedSelector(feature)) {
144             if (isDefined(feature.propertyKey())) {
145                 definitionError.accept(
146                         new IllegalStateException("propertyKey must not be set if selector is set on bean " + bean));
147             }
148             validateSelectorType(feature.selector(), bean, beanManager, definitionError);
149         } else if (isDefined(feature.propertyKey()) && !mpConfigAvailable) {
150             definitionError.accept(new IllegalStateException(
151                     "as MicroProfile Config is not available, propertyKey must not be set on bean " + bean));
152         } else if (!isDefined(feature.propertyKey()) && bean.getTypes().stream().map(FeaturesExtension::toClass)
153                 .noneMatch(Selector.class::isAssignableFrom)) {
154                     definitionError.accept(new IllegalStateException(
155                             bean + " must implement " + Selector.class.getName() + " if no other selector is defined"));
156                 }
157         if (isDefined(feature.propertyValue()) && !isDefined(feature.propertyKey())) {
158             definitionError.accept(new IllegalStateException(
159                     "propertyValue must not be set if propertyKey is not set on bean " + bean));
160         }
161         validateCacheDurationMillisProperty(feature, bean, definitionError);
162     }

164     private void validateSelectorType(Class<? extends ContextualSelector<?>> selector, Bean<?> bean,
165             BeanManager beanManager, Consumer<Throwable> definitionError) {
166         beanManager.createAnnotatedType(selector).getTypeClosure()
167                 .stream()
168                 .filter(type -> type instanceof ParameterizedType parametrizedType
169                         && ContextualSelector.class.equals(parametrizedType.getRawType()))
170                 .map(ParameterizedType.class::cast) //
171                 .map(type -> type.getActualTypeArguments()[0]) //
172                 .filter(type -> !bean.getTypes().contains(type)) //
173                 .forEach(type -> definitionError.accept(new IllegalStateException("selector type " + selector.getName()
174                         + " accepts beans with type " + type.getTypeName() + ", which is not a type of the bean " + bean)));
175     }

177     private void validateCacheDurationMillisProperty(Feature feature, Bean<?> bean,
178             Consumer<Throwable> definitionError) {
179         if (!mpConfigAvailable && isDefined(feature.cache().durationMillisProperty())) {
180             definitionError.accept(new IllegalStateException(
181                     "as MicroProfile Config is not available, cache durationMillisProperty must not be set on bean "
182                             + bean));
183         }
184     }

186     private static Class<?> toClass(Type type) {
187         if (type instanceof Class) {
188             return (Class<?>) type;
189         }
190         if (type instanceof ParameterizedType parameterizedType) {
191             return toClass(parameterizedType.getRawType());
192         }
193         throw new IllegalStateException("unsupported type " + type);
194     }

196     void registerFeatureSelectorBeans(@Priority(LIBRARY_AFTER + 500) @Observes AfterBeanDiscovery abd, BeanManager beanManager) {
197         for (Entry<Type, Set<Bean<?>>> feature : getFeatures().entrySet()) {
198             if (beanManager.getBeans(feature.getKey()).isEmpty()) {
199                 BeanConfigurator<Object> configurator = abd.addBean().types(feature.getKey());
200                 if (useInterceptor) {
201                     configurator.createWith(
202                             ctx -> createIntercepted(beanManager, feature.getKey(), ctx, feature.getValue()));
203                 } else {
204                     configurator //
205                             .scope(FeatureScoped.class) //
206                             .createWith(ctx -> createDummy(beanManager, feature.getKey(), ctx, feature.getValue()));
207                 }
208             }
209         }
210     }

212     private Map<Type, Set<Bean<?>>> getFeatures() {
213         Map<Type, Set<Bean<?>>> features = new HashMap<>();
214         for (Bean<?> featureBean : featureBeans) {
215             for (Type type : featureBean.getTypes()) {
216                 if (!Object.class.equals(type) && !ContextualSelector.class.equals(type)
217                         && !(type instanceof ParameterizedType parametrizedType
218                                 && ContextualSelector.class.equals(parametrizedType.getRawType()))
219                         && !Selector.class.equals(type) && !ThrowableSelector.class.equals(type)) {
220                     features.computeIfAbsent(type, t -> new HashSet<>()).add(featureBean);
221                 }
222             }
223         }
224         return features;
225     }

227     private <T> T createIntercepted(BeanManager beanManager, Type type, CreationalContext<Object> ctx,
228             Set<Bean<? extends T>> value) {
229         // register target beans
230         invokerFactory.addInstanceFactory(beanManager, type,
231                 creationContext -> getFeatureInstances(beanManager, type, creationContext, value));
232         @SuppressWarnings("unchecked")
233         InterceptionFactory<T> interceptionFactory = beanManager.createInterceptionFactory(ctx,
234                 value.stream().filter(invokerFactory::isManagedBean).map(Bean::getBeanClass).map(Class.class::cast)
235                         .findFirst().orElseGet(() -> toClass(type)));
236         interceptionFactory.configure().add(FeatureSelector.Literal.INSTANCE);
237         return interceptionFactory.createInterceptedInstance(null);
238     }

240     private <T> T createDummy(BeanManager beanManager, Type type, CreationalContext<Object> ctx,
241             Set<Bean<? extends T>> features) {
242         Map<Bean<? extends T>, Supplier<T>> instances = getFeatureInstances(beanManager, type, ctx, features);
243         @SuppressWarnings("unchecked")
244         Bean<T> ownBean = (Bean<T>) beanManager.resolve(beanManager.getBeans(type));
245         context.setInstances(ownBean, new FeatureInstances<>(instances, getSelectors(beanManager, ctx, instances),
246                 getCache(beanManager, ctx)));
247         return instances.values().iterator().next().get();
248     }

255     private static <T> Map<Bean<? extends T>, Supplier<T>> getFeatureInstances(BeanManager beanManager, Type type,
256             CreationalContext<?> ctx, Set<Bean<? extends T>> features) {
257         return features.stream().collect(toMap(identity(), bean -> {
258             @SuppressWarnings("unchecked")
259             T instance = (T) beanManager.getReference(bean, type, ctx);
260             return () -> instance;
261         }));
262     }

264     private static <T> Map<Bean<? extends T>, ContextualSelector<? super T>> getSelectors(BeanManager beanManager,
265             CreationalContext<?> ctx, Map<Bean<? extends T>, Supplier<T>> instances) {
266         Map<Bean<? extends T>, ContextualSelector<? super T>> selectors = new HashMap<>();
267         for (Entry<Bean<? extends T>, Supplier<T>> instance : instances.entrySet()) {
268             Feature feature = feature(instance.getKey()).orElseThrow();
269             ContextualSelector<?> selector;
270             if (feature.remaining()) {
271                 selector = null;
272             } else if (hasDefinedSelector(feature)) {
273                 Set<Bean<?>> beans = beanManager.getBeans(feature.selector());
274                 if (beans.isEmpty()) {
275                     selector = createInstance(feature.selector());
276                 } else {
277                     selector = (ContextualSelector<?>) beanManager.getReference(beanManager.resolve(beans),
278                             GENERIC_CONTEXTUAL_SELECTOR_TYPE, ctx);
279                 }
280             } else if (isDefined(feature.propertyKey())) {
281                 selector = (ContextualSelector<?>) beanManager.getReference(
282                         beanManager.resolve(beanManager.getBeans(ConfigurationSelector.class)),
283                         ContextualSelector.class, ctx);
284             } else {
285                 selector = (Selector) instance.getValue().get();
286             }
287             @SuppressWarnings("unchecked") // is checked in the validate method
288             ContextualSelector<? super T> contextualSelector = (ContextualSelector<? super T>) selector;
289             selectors.put(instance.getKey(), contextualSelector);
290         }
291         return selectors;
292     }

294     static <T> T createInstance(Class<T> clazz) {
295         try {
296             return clazz.getConstructor().newInstance();
297         } catch (InstantiationException | IllegalAccessException | InvocationTargetException
298                 | NoSuchMethodException e) {
299             throw new IllegalStateException("Failed to create selector pojo instance " + clazz, e);
300         }
301     }

303     static boolean hasDefinedSelector(Feature feature) {
304         return !Selector.class.equals(feature.selector());

307     void registerContext(@Observes AfterBeanDiscovery abd) {
308         abd.addContext(context);
309         try {
310             Class<?> shutdownEvent = Class.forName("jakarta.enterprise.event.Shutdown", false,
311                     getClass().getClassLoader());
312             abd.addObserverMethod() //
313                     .observedType(shutdownEvent) //
314                     .priority(LIBRARY_AFTER + 900) //
315                     .notifyWith(event -> context.stop());
316         } catch (ClassNotFoundException e) {
317             // ignore
318         }
319     }

330     private static class FeatureInvokerFactory {
331         private Map<Bean<?>, InstancesFactory<?>> instancesFactory = new ConcurrentHashMap<>();
332         private Set<Type> registeredTypes = ConcurrentHashMap.newKeySet();
333         private Set<Bean<?>> managedBeans = ConcurrentHashMap.newKeySet();
334 
335         public <T> void processFeatureBean(ProcessBean<T> event) {
336             if (event instanceof ProcessManagedBean<T>) {
337                 managedBeans.add(event.getBean());
338             }
339         }
340 
341         public boolean isManagedBean(Bean<?> bean) {
342             return managedBeans.contains(bean);
343         }
344 
345         public <T> void addInstanceFactory(BeanManager beanManager, Type type, InstancesFactory<T> factory) {
346             if (registeredTypes.add(type)) {
347                 instancesFactory.putIfAbsent(beanManager.resolve(beanManager.getBeans(type)), factory);
348             }
349         }
350 
351         public <T> FeatureInvoker<T> createInvoker(CreationalContext<FeatureInterceptor> ctx, Bean<T> targetBean,
352                 BeanManager beanManager) {
353             Map<Bean<? extends T>, Supplier<T>> instances = getInstances(ctx, targetBean);
354             return new FeatureInvoker<>(targetBean, instances, getSelectors(beanManager, ctx, instances),
355                     getCache(beanManager, ctx));
356         }
357 
358         @SuppressWarnings("unchecked")
359         protected <T> Map<Bean<? extends T>, Supplier<T>> getInstances(CreationalContext<FeatureInterceptor> ctx,
360                 Bean<T> targetBean) {
361             return ((InstancesFactory<T>) instancesFactory.get(targetBean)).create(ctx);
362         }
363     }

365     private static class FeatureInvokerInvokerFactory extends FeatureInvokerFactory {
366         private Map<Bean<?>, Map<Method, ? extends Invoker<?, ?>>> invokers = new HashMap<>();
367 
368         public FeatureInvokerInvokerFactory() {
369             Invoker.class.getName(); // ensure that the Invoker class is available
370         }
371 
372         @Override
373         public <T> void processFeatureBean(ProcessBean<T> event) {
374             super.processFeatureBean(event);
375             if (event instanceof ProcessManagedBean<T> processManagedBean) {
376                 Map<Method, Invoker<?, ?>> beanInvokers = new ConcurrentHashMap<>();
377                 processManagedBean.getAnnotatedBeanClass().getMethods().stream().filter(method -> !method.isStatic())
378                         .forEach(method -> beanInvokers.put(method.getJavaMember(),
379                                 processManagedBean.createInvoker(method).build()));
380                 invokers.put(processManagedBean.getBean(), beanInvokers);
381             }
382         }
383 
384         @Override
385         public <T> FeatureInvoker<T> createInvoker(CreationalContext<FeatureInterceptor> ctx, Bean<T> targetBean,
386                 BeanManager beanManager) {
387             Map<Bean<? extends T>, Supplier<T>> instances = getInstances(ctx, targetBean);
388             return new FeatureInvokerInvoker<>(targetBean, instances, getSelectors(beanManager, ctx, instances),
389                     getCache(beanManager, ctx), invokers);
390         }
391     }
