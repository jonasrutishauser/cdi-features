3   import java.lang.reflect.Method;
4   import java.util.Arrays;
5   import java.util.Map;
6   import java.util.function.Supplier;
7   
8   import io.github.jonasrutishauser.cdi.features.ContextualSelector;
9   import jakarta.enterprise.inject.spi.Bean;
10  import jakarta.enterprise.invoke.Invoker;
11  import jakarta.interceptor.InvocationContext;
12  
13  class FeatureInvokerInvoker<T> extends FeatureInvoker<T> {
14  
15      private Map<Bean<?>, Map<Method, ? extends Invoker<?, ?>>> invokers;
16  
17      public FeatureInvokerInvoker(Bean<T> targetBean, Map<Bean<? extends T>, Supplier<T>> instances,
18              Map<Bean<? extends T>, ContextualSelector<? super T>> selectors, Cache cache,
19              Map<Bean<?>, Map<Method, ? extends Invoker<?, ?>>> invokers) {
20          super(targetBean, instances, selectors, cache);
21          this.invokers = invokers;
22      }
23  
24      @Override
25      protected Object invoke(InvocationContext context, Bean<?> selectedBean, T target) throws Exception {
26          @SuppressWarnings("unchecked")
27          Map<Method, Invoker<T, ?>> selectedInvokers = (Map<Method, Invoker<T, ?>>) invokers.get(selectedBean);
28          if (selectedInvokers != null) {
29              Method method = context.getMethod();
30              Invoker<T, ?> invoker = selectedInvokers.get(method);
31              if (invoker == null) {
32                  invoker = selectedInvokers.entrySet().stream() //
33                          .filter(entry -> entry.getKey().getName().equals(method.getName())
34                                  && Arrays.equals(entry.getKey().getParameterTypes(), method.getParameterTypes())
35                                  && entry.getKey().getReturnType().equals(method.getReturnType())) //
36                          .map(Map.Entry::getValue) //
37                          .findFirst() //
38                          .orElseThrow(() -> new IllegalStateException("No method found: " + method));
39              }
40              return invoker.invoke(target, context.getParameters());
41          }
42          return super.invoke(context, selectedBean, target);
43      }
