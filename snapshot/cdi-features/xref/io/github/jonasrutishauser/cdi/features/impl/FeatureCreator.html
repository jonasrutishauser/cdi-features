28  public class FeatureCreator implements SyntheticBeanCreator<Object>  {
29  
30      static final String IDENTIFIER = "identfier";
31      static final String BEANS_IDENTIFIER = "beans.identfier";
32  
33      @FunctionalInterface
34      interface Lookup {
35          <T> T getReference(Class<T> type) throws IllegalArgumentException;
36      }
37  
38      static <T> T create(Bean<T> ownBean, Stream<? extends Handle<? extends T>> instancesStream, Lookup lookup) {
39          Map<Bean<? extends T>, Supplier<T>> instances = instancesStream
40                  .collect(Collectors.toMap(Handle::getBean, handle -> handle::get));
41          Map<Bean<? extends T>, ContextualSelector<? super T>> selectors = getSelectors(lookup, instances);
42          FeatureContext context = (FeatureContext) lookup.getReference(BeanManager.class)
43                  .getContext(FeatureScoped.class);
44          context.setInstances(ownBean, new FeatureInstances<>(instances, selectors, lookup.getReference(Cache.class)));
45          return instances.values().iterator().next().get();
46      }
47  
48      private static <T> Map<Bean<? extends T>, ContextualSelector<? super T>> getSelectors(Lookup lookup,
49              Map<Bean<? extends T>, Supplier<T>> instances) {
50          Map<Bean<? extends T>, ContextualSelector<? super T>> selectors = new HashMap<>();
51          for (Entry<Bean<? extends T>, Supplier<T>> instance : instances.entrySet()) {
52              Feature feature = feature(instance.getKey()).orElseThrow();
53              ContextualSelector<?> selector;
54              if (feature.remaining()) {
55                  selector = null;
56              } else if (hasDefinedSelector(feature)) {
57                  try {
58                      selector = lookup.getReference(feature.selector());
59                  } catch (IllegalArgumentException e) {
60                      selector = createInstance(feature.selector());
61                  }
62              } else if (isDefined(feature.propertyKey())) {
63                  selector = lookup.getReference(ConfigurationSelector.class);
64              } else {
65                  selector = ctx -> ((Selector) instance.getValue().get()).selected(ctx);
66              }
67              @SuppressWarnings("unchecked") // is checked by the extension
68              ContextualSelector<? super T> contextualSelector = (ContextualSelector<? super T>) selector;
69              selectors.put(instance.getKey(), contextualSelector);
70          }
71          return selectors;
72      }
73  
74      @Override
75      public Object create(Instance<Object> lookup, Parameters params) {
76          BeanContainer beanContainer = lookup.select(BeanContainer.class).get();
77          @SuppressWarnings("unchecked")
78          Bean<Object> ownBean = (Bean<Object>) beanContainer
79                  .resolve(beanContainer.getBeans(Object.class, params.get(IDENTIFIER, Annotation.class)));
80          return create(ownBean, //
81                  Arrays.stream(params.get(BEANS_IDENTIFIER, Annotation[].class)) //
82                          .map(lookup::select) //
83                          .map(Instance::getHandle), //
84                  new Lookup() {
85                      @Override
86                      public <T> T getReference(Class<T> type) throws IllegalArgumentException {
87                          Instance<T> instance = lookup.select(type);
88                          if (!instance.isResolvable()) {
89                              throw new IllegalArgumentException("No bean found for required type [" + type + "]");
90                          }
91                          return instance.get();
92                      }
93                  });
94      }
