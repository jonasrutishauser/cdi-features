63      @BuildStep
64      void improveFeatureBeans(BeanDiscoveryFinishedBuildItem discoveryFinished,
65              BeanRegistrationPhaseBuildItem beanRegistrationPhase, BuildProducer<SyntheticBeanBuildItem> syntheticBean,
66              BuildProducer<GeneratedClassBuildItem> generatedClass) {
67          AnnotationLiteralProcessor annotationLiteralProcessor = beanRegistrationPhase.getBeanProcessor()
68                  .getAnnotationLiteralProcessor();
69          Type objectType = Type.create(Object.class);
70          AnnotationInstance featureAnnotation = AnnotationInstance.builder(Feature.class).build();
71          Optional<ClassInfo> featureClass = beanRegistrationPhase.getBeanProcessor().getBeanDeployment().getQualifiers()
72                  .stream().filter(c -> DotName.createSimple(Feature.class).equals(c.name())).findAny();
73          GeneratedClassGizmoAdaptor classGizmoAdaptor = new GeneratedClassGizmoAdaptor(generatedClass, true);
74          Map<String, Integer> counter = new HashMap<>();
75          discoveryFinished.beanStream().withQualifier(Identified.class).withScope(FeatureScoped.class).forEach(bean -> {
76              Type beanType = bean.getTypes().stream().filter(t -> !objectType.equals(t)).findAny().get();
77              ParameterizedType instanceType = ParameterizedType.create(Instance.class, beanType);
78              String classNamePrefix = beanType.toString();
79              if (classNamePrefix.contains("<")) {
80                  classNamePrefix = classNamePrefix.substring(0, classNamePrefix.indexOf('<'));
81              }
82              String instanceTypeLiteralName = classNamePrefix + "$$instanceTypeLiteral$$"
83                      + counter.compute(classNamePrefix, (key, value) -> value == null ? 0 : value++);
84              ClassCreator.builder().classOutput(classGizmoAdaptor).className(instanceTypeLiteralName)
85                      .signature(SignatureBuilder.forClass()
86                              .setSuperClass(toGizmo(ParameterizedType.create(TypeLiteral.class, instanceType))))
87                      .build().close();
88              ExtendedBeanConfigurator configurator = SyntheticBeanBuildItem.configure(bean.getBeanClass()) //
89                      .types(bean.getTypes().toArray(Type[]::new)) //
90                      .scope(FeatureScoped.class) //
91                      .addInjectionPoint(instanceType, featureAnnotation) //
92                      .addInjectionPoint(Type.create(BeanManager.class)) //
93                      .addInjectionPoint(Type.create(DotName.createSimple(CACHE_CLASS), Kind.CLASS)) //
94                      .creator(mc -> {
95                          MethodDescriptor create = MethodDescriptor.ofMethod(ArcFeatureCreator.class, "create",
96                                  Object.class, SyntheticCreationalContext.class, TypeLiteral.class, Bean.class,
97                                  Feature.class);
98                          MethodDescriptor instanceLiteralConstructor = MethodDescriptor
99                                  .ofConstructor(instanceTypeLiteralName);
100                         mc.returnValue(mc.invokeStaticMethod(create, //
101                                 mc.getMethodParam(0), //
102                                 mc.newInstance(instanceLiteralConstructor), //
103                                 mc.getThis(),
104                                 annotationLiteralProcessor.create(mc, featureClass.get(), featureAnnotation)));
105                     });
106             addSelectors(configurator, discoveryFinished, beanType, featureAnnotation);
107             syntheticBean.produce(configurator.done());
108         });
109     }
110 
111     private void addSelectors(ExtendedBeanConfigurator configurator, BeanDiscoveryFinishedBuildItem discoveryFinished,
112             Type beanType, AnnotationInstance featureAnnotation) {
113         Type selectorClass = Type.create(Selector.class);
114         boolean configurationSelectorAdded = false;
115         for (BeanInfo featureBean : discoveryFinished.beanStream().assignableTo(beanType, featureAnnotation)
116                 .collect()) {
117             AnnotationInstance feature = featureBean.getQualifier(DotName.createSimple(Feature.class)).orElseThrow();
118             AnnotationValue selectorValue = feature.value("selector");
119             AnnotationValue propertyKeyValue = feature.value("propertyKey");
120             if (selectorValue != null && !selectorClass.equals(selectorValue.asClass())) {
121                 if (!discoveryFinished.beanStream().assignableTo(selectorValue.asClass()).isEmpty()) {
122                     configurator.addInjectionPoint(selectorValue.asClass());
123                 }
124             } else
125                 if (!configurationSelectorAdded && propertyKeyValue != null && !propertyKeyValue.asString().isEmpty()) {
126                     configurator.addInjectionPoint(
127                             Type.create(DotName.createSimple(CONFIGURATION_SELECTOR_CLASS), Kind.CLASS));
128                     configurationSelectorAdded = true;
129                 }
130         }
131     }

133     private io.quarkus.gizmo.Type.ParameterizedType toGizmo(ParameterizedType parameterizedType) {
134         return io.quarkus.gizmo.Type.parameterizedType(io.quarkus.gizmo.Type.classType(parameterizedType.name()),
135                 parameterizedType.arguments().stream().map(type -> {
136                     if (type.kind() == Kind.PARAMETERIZED_TYPE) {
137                         return toGizmo(type.asParameterizedType());
138                     } else {
139                         return io.quarkus.gizmo.Type.classType(type.name());
140                     }
141                 }).toArray(io.quarkus.gizmo.Type[]::new));
142     }
