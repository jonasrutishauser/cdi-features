86      @BuildStep(onlyIf = UseInterceptor.class)
87      void useInvocationContextTargetIfPossible(CombinedIndexBuildItem index,
88              BuildProducer<GeneratedClassBuildItem> generatedClass,
89              BuildProducer<BytecodeTransformerBuildItem> bytcodeTransformer, CdiFeaturesConfiguration configuration) {
90          ClassInfo invocationContext = index.getComputingIndex()
91                  .getClassByName("io.quarkus.arc.impl.AbstractInvocationContext");
92          FieldInfo targetField = invocationContext == null ? null : invocationContext.field("target");
93          if (targetField == null) {
94              LOG.warn("Could not find target field in AbstractInvocationContext, intercepted features will use reflection");
95          } else {
96              GeneratedClassGizmoAdaptor classGizmoAdaptor = new GeneratedClassGizmoAdaptor(generatedClass, false);
97              String className = "io.quarkus.arc.impl.InvocationContext_Target_Setter";
98              MethodDescriptor setTargetMethod = MethodDescriptor.ofMethod(className, "setTarget", void.class,
99                      InvocationContext.class, Object.class);
100             try (ClassCreator setterClass = ClassCreator.builder().classOutput(classGizmoAdaptor).className(className)
101                     .build()) {
102                 MethodCreator setter = setterClass.getMethodCreator(setTargetMethod)
103                         .setModifiers(Opcodes.ACC_STATIC | Opcodes.ACC_PUBLIC);
104                 setter.writeInstanceField(FieldDescriptor.of(targetField), setter.getMethodParam(0),
105                         setter.getMethodParam(1));
106                 setter.returnNull();
107             }
108 
109             // change invoke method to use the setter and then call proceed()
110             ClassTransformer ct = new ClassTransformer(FEATURE_INVOKER_CLASS);
111             MethodDescriptor invokeDescriptor = MethodDescriptor.ofMethod(FEATURE_INVOKER_CLASS, "invoke", Object.class,
112                     InvocationContext.class, Bean.class, Object.class);
113             ct.removeMethod(invokeDescriptor);
114 
115             MethodCreator invokeMethod = ct.addMethod(invokeDescriptor).setModifiers(Opcodes.ACC_PROTECTED)
116                     .addException(Exception.class);
117             // set the target
118             invokeMethod.invokeStaticMethod(setTargetMethod, invokeMethod.getMethodParam(0),
119                     invokeMethod.getMethodParam(2));
120             // call proceed
121             invokeMethod.returnValue(invokeMethod.invokeInterfaceMethod(
122                     MethodDescriptor.ofMethod(InvocationContext.class, "proceed", Object.class),
123                     invokeMethod.getMethodParam(0)));
124 
125             bytcodeTransformer.produce(new BytecodeTransformerBuildItem.Builder().setCacheable(true)
126                     .setClassToTransform(FEATURE_INVOKER_CLASS)
127                     .setVisitorFunction((ignored, visitor) -> ct.applyTo(visitor)).build());
128         }
129     }

131     @BuildStep
132     void improveFeatureBeans(BeanDiscoveryFinishedBuildItem discoveryFinished,
133             BeanRegistrationPhaseBuildItem beanRegistrationPhase, BuildProducer<SyntheticBeanBuildItem> syntheticBean,
134             BuildProducer<GeneratedClassBuildItem> generatedClass, CdiFeaturesConfiguration configuration) {
135         AnnotationLiteralProcessor annotationLiteralProcessor = beanRegistrationPhase.getBeanProcessor()
136                 .getAnnotationLiteralProcessor();
137         Type objectType = Type.create(Object.class);
138         AnnotationInstance featureAnnotation = AnnotationInstance.builder(Feature.class).build();
139         Optional<ClassInfo> featureClass = beanRegistrationPhase.getBeanProcessor().getBeanDeployment().getQualifiers()
140                 .stream().filter(c -> DotName.createSimple(Feature.class).equals(c.name())).findAny();
141         GeneratedClassGizmoAdaptor classGizmoAdaptor = new GeneratedClassGizmoAdaptor(generatedClass, true);
142         Map<String, Integer> counter = new HashMap<>();
143         discoveryFinished.beanStream().withQualifier(Identified.class).withScope(FeatureScoped.class).forEach(bean -> {
144             Type beanType = bean.getTypes().stream().filter(t -> !objectType.equals(t)).findAny().get();
145             ParameterizedType instanceType = ParameterizedType.create(Instance.class, beanType);
146             String classNamePrefix = beanType.toString();
147             if (classNamePrefix.contains("<")) {
148                 classNamePrefix = classNamePrefix.substring(0, classNamePrefix.indexOf('<'));
149             }
150             String instanceTypeLiteralName = classNamePrefix + "$$instanceTypeLiteral$$"
151                     + counter.compute(classNamePrefix, (key, value) -> value == null ? 0 : ++value);
152             ClassCreator.builder().classOutput(classGizmoAdaptor).className(instanceTypeLiteralName)
153                     .signature(SignatureBuilder.forClass()
154                             .setSuperClass(toGizmo(ParameterizedType.create(TypeLiteral.class, instanceType))))
155                     .build().close();
156             ExtendedBeanConfigurator configurator = SyntheticBeanBuildItem.configure(bean.getBeanClass()) //
157                     .types(bean.getTypes().toArray(Type[]::new)) //
158                     .addInjectionPoint(instanceType, featureAnnotation) //
159                     .addInjectionPoint(Type.create(DotName.createSimple(CACHE_CLASS), Kind.CLASS));
160             Class<?> creatorClass;
161             if (configuration.useInterceptor()) {
162                 creatorClass = ArcInterceptedFeatureCreator.class;
163                 configurator.injectInterceptionProxy(
164                         DotName.createSimple(ArcInterceptedFeatureCreator.BINDING_SOURCE_CLASS));
165             } else {
166                 creatorClass = ArcFeatureCreator.class;
167                 configurator.scope(FeatureScoped.class).addInjectionPoint(Type.create(BeanManager.class));
168             }
169             configurator.creator(mc -> {
170                 MethodDescriptor create = MethodDescriptor.ofMethod(creatorClass, "create", Object.class,
171                         SyntheticCreationalContext.class, TypeLiteral.class, Bean.class, Feature.class);
172                 MethodDescriptor instanceLiteralConstructor = MethodDescriptor.ofConstructor(instanceTypeLiteralName);
173                 mc.returnValue(mc.invokeStaticMethod(create, //
174                         mc.getMethodParam(0), //
175                         mc.newInstance(instanceLiteralConstructor), //
176                         mc.getThis(), annotationLiteralProcessor.create(mc, featureClass.get(), featureAnnotation)));
177             });
178             addSelectors(configurator, discoveryFinished, beanType, featureAnnotation);
179             syntheticBean.produce(configurator.done());
180         });
181     }

183     private void addSelectors(ExtendedBeanConfigurator configurator, BeanDiscoveryFinishedBuildItem discoveryFinished,
184             Type beanType, AnnotationInstance featureAnnotation) {
185         Type selectorClass = Type.create(Selector.class);
186         boolean configurationSelectorAdded = false;
187         for (BeanInfo featureBean : discoveryFinished.beanStream().assignableTo(beanType, featureAnnotation)
188                 .collect()) {
189             AnnotationInstance feature = featureBean.getQualifier(DotName.createSimple(Feature.class)).orElseThrow();
190             AnnotationValue selectorValue = feature.value("selector");
191             AnnotationValue propertyKeyValue = feature.value("propertyKey");
192             if (selectorValue != null && !selectorClass.equals(selectorValue.asClass())) {
193                 if (!discoveryFinished.beanStream().assignableTo(selectorValue.asClass()).isEmpty()) {
194                     configurator.addInjectionPoint(selectorValue.asClass());
195                 }
196             } else
197                 if (!configurationSelectorAdded && propertyKeyValue != null && !propertyKeyValue.asString().isEmpty()) {
198                     configurator.addInjectionPoint(
199                             Type.create(DotName.createSimple(CONFIGURATION_SELECTOR_CLASS), Kind.CLASS));
200                     configurationSelectorAdded = true;
201                 }
202         }
203     }

205     private io.quarkus.gizmo.Type.ParameterizedType toGizmo(ParameterizedType parameterizedType) {
206         return io.quarkus.gizmo.Type.parameterizedType(io.quarkus.gizmo.Type.classType(parameterizedType.name()),
207                 parameterizedType.arguments().stream().map(type -> {
208                     if (type.kind() == Kind.PARAMETERIZED_TYPE) {
209                         return toGizmo(type.asParameterizedType());
210                     } else {
211                         return io.quarkus.gizmo.Type.classType(type.name());
212                     }
213                 }).toArray(io.quarkus.gizmo.Type[]::new));
214     }
