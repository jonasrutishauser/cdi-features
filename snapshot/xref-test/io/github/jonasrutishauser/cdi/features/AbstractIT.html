124     @ApplicationScoped
125     static class Config {
126         private int selected;
127         private boolean destroyed;
128         private boolean selectorDestroyed;
129         private boolean feature3Created;
130 
131         public void setSelected(int selected) {
132             this.selected = selected;
133             destroyed = selectorDestroyed = feature3Created = false;
134         }
135 
136         public int getSelected() {
137             return selected;
138         }
139 
140         public boolean isDestroyed() {
141             return destroyed && selectorDestroyed;
142         }
143 
144         public boolean isFeature3Created() {
145             return feature3Created;
146         }
147 
148         public void setDestroyed() {

150         }
151 
152         public void setSelectorDestroyed() {
153             this.selectorDestroyed = true;
154         }
