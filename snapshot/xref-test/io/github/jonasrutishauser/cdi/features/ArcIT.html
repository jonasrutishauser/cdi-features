17      @Override
18      protected void setSelected(int selected) {
19          try {
20              destroy(instance.select(Class.forName(getClass().getPackage().getName() + ".impl.Cache"))); // clear cache
21          } catch (ClassNotFoundException e) {
22              throw new IllegalStateException(e);
23          }
24          super.setSelected(selected);
25      }
26  
27      private <T> void destroy(Instance<T> instance) {
28          instance.destroy(instance.get());
29      }
