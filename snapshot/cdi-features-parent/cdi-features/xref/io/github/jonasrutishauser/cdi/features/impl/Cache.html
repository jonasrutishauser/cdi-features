3   import java.time.Clock;
4   import java.time.Instant;
5   import java.util.Collection;
6   import java.util.concurrent.ConcurrentHashMap;
7   import java.util.concurrent.ConcurrentMap;
8   import java.util.function.Function;
9   import java.util.function.ToLongFunction;
10  
11  import io.github.jonasrutishauser.cdi.features.NoSelectedFeatureException;
12  import jakarta.enterprise.context.spi.Contextual;
13  import jakarta.enterprise.inject.spi.Bean;
14  
15  class Cache {
16  
17      private final ConcurrentMap<Contextual<?>, CacheEntry> cache = new ConcurrentHashMap<>();
18      private final Clock clock = Clock.systemUTC();
19  
20      public <T> Bean<?> compute(Contextual<T> key, Collection<Bean<? extends T>> items,
21              Function<Bean<? extends T>, Boolean> selection, ToLongFunction<Bean<? extends T>> cacheDurationInMillis) {
22          CacheEntry entry = cache.compute(key, (k, current) -> {
23              if (current != null && (current.forever || !current.validUntil.isBefore(Instant.now(clock)))) {
24                  return current;
25              }
26              Bean<? extends T> remaining = null;
27              for (Bean<? extends T> item : items) {
28                  Boolean selected = selection.apply(item);
29                  if (selected == null) {
30                      remaining = item;
31                  } else if (selected.booleanValue()) {
32                      return new CacheEntry(item, validUntil(cacheDurationInMillis.applyAsLong(item)));
33                  }
34              }
35              if (remaining == null) {
36                  throw new NoSelectedFeatureException(k);
37              }
38              return new CacheEntry(remaining, validUntil(cacheDurationInMillis.applyAsLong(remaining)));
39          });
40          if (entry.validUntil.isBefore(Instant.now(clock))) {
41              cache.remove(key, entry);
42          }
43          return entry.target;
44      }
45  
46      private Instant validUntil(long durationInMillis) {
47          if (durationInMillis < 0) {
48              return Instant.MAX;
49          } else if (durationInMillis == 0) {
50              return Instant.MIN;
51          }
52          return Instant.now(clock).plusMillis(durationInMillis);
53      }
54  
55      private static class CacheEntry {
56          final Bean<?> target;
57          final Instant validUntil;
58          final boolean forever;
59  
60          public CacheEntry(Bean<?> target, Instant validUntil) {
61              this.target = target;
62              this.validUntil = validUntil;
63              this.forever = validUntil.equals(Instant.MAX);
64          }
65      }
