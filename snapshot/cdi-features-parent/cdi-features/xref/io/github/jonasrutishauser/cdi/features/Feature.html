5   import java.lang.annotation.Documented;
6   import java.lang.annotation.Retention;
7   import java.lang.annotation.Target;
8   
9   import jakarta.enterprise.util.Nonbinding;
10  import jakarta.inject.Qualifier;
11  
12  @Qualifier
13  @Documented
14  @Retention(RUNTIME)
15  public @interface Feature {
16  
17      /**
18       * @return selector class which will be lookup up as a CDI bean if possible
19       *         (needs a public no-arg constructor otherwise).
20       */
21      @Nonbinding
22      Class<? extends ContextualSelector> selector() default Selector.class;
23  
24      /**
25       * @return property key used for the selection. This requires that MicroProfile
26       *         Config is available.
27       */
28      @Nonbinding
29      String propertyKey() default "";
30  
31      /**
32       * @return property value which is expected for the selection. If not specified
33       *         any non null value is accepted.
34       */
35      @Nonbinding
36      String propertyValue() default "";
37  
38      /**
39       * @return {@code true} if it should be used if no other bean is selected. There
40       *         can be at most one for a given type.
41       */
42      @Nonbinding
43      boolean remaining() default false;
44  
45      /**
46       * @return the cache configuration if this bean is selected.
47       */
48      @Nonbinding
49      Cache cache() default @Cache;
50  
51      @Target({})
52      @interface Cache {
53          /**
54           * @return the cache duration in milliseconds. The value {@code 0} means no
55           *         caching and any negative value means forever.
56           */
57          long durationMillis() default 0;
58  
59          /**
60           * @return the property key which defines the duration. It is possible to use
61           *         <code>@{type}</code> to reference the injected type.
62           * @see #durationMillis()
63           */
64          String durationMillisProperty() default "";
65      }
66  }
