29      static final String IDENTIFIER = "identfier";
30      static final String BEANS_IDENTIFIER = "beans.identfier";
31  
32      @FunctionalInterface
33      interface Lookup {
34          <T> T getReference(Class<T> type) throws IllegalArgumentException;
35      }
36  
37      static <T> T create(Bean<T> ownBean, Stream<? extends Handle<? extends T>> instancesStream, Lookup lookup) {
38          Map<Bean<? extends T>, T> instances = instancesStream.collect(Collectors.toMap(Handle::getBean, Handle::get));
39          Map<Bean<? extends T>, ContextualSelector> selectors = getSelectors(lookup, instances);
40          FeatureContext context = (FeatureContext) lookup.getReference(BeanManager.class)
41                  .getContext(FeatureScoped.class);
42          context.setInstances(ownBean, new FeatureInstances<>(instances, selectors, lookup.getReference(Cache.class)));
43          return instances.values().iterator().next();
44      }
45  
46      private static <T> Map<Bean<? extends T>, ContextualSelector> getSelectors(Lookup lookup,
47              Map<Bean<? extends T>, T> instances) {
48          Map<Bean<? extends T>, ContextualSelector> selectors = new HashMap<>();
49          for (Entry<Bean<? extends T>, T> instance : instances.entrySet()) {
50              Feature feature = feature(instance.getKey()).orElseThrow();
51              ContextualSelector selector;
52              if (feature.remaining()) {
53                  selector = null;
54              } else if (hasDefinedSelector(feature)) {
55                  try {
56                      selector = lookup.getReference(feature.selector());
57                  } catch (IllegalArgumentException e) {
58                      selector = createInstance(feature.selector());
59                  }
60              } else if (isDefined(feature.propertyKey())) {
61                  selector = lookup.getReference(ConfigurationSelector.class);
62              } else {
63                  selector = (Selector) instance.getValue();
64              }
65              selectors.put(instance.getKey(), selector);
66          }
67          return selectors;
68      }

70      @Override
71      public Object create(Instance<Object> lookup, Parameters params) {
72          BeanContainer beanContainer = lookup.select(BeanContainer.class).get();
73          @SuppressWarnings("unchecked")
74          Bean<Object> ownBean = (Bean<Object>) beanContainer
75                  .resolve(beanContainer.getBeans(Object.class, params.get(IDENTIFIER, Annotation.class)));
76          return create(ownBean, //
77                  Arrays.stream(params.get(BEANS_IDENTIFIER, Annotation[].class)) //
78                          .map(lookup::select) //
79                          .map(Instance::getHandle), //
80                  new Lookup() {
81                      @Override
82                      public <T> T getReference(Class<T> type) throws IllegalArgumentException {
83                          Instance<T> instance = lookup.select(type);
84                          if (!instance.isResolvable()) {
85                              throw new IllegalArgumentException("No bean found for required type [" + type + "]");
86                          }
87                          return instance.get();
88                      }
89                  });
90      }
