21  class FeatureInstances<T> {
22      private final Map<Bean<? extends T>, T> instances;
23      private final Map<Bean<? extends T>, ContextualSelector> selectors;
24      private final Cache cache;
25  
26      public FeatureInstances(Map<Bean<? extends T>, T> instances, Map<Bean<? extends T>, ContextualSelector> selectors,
27              Cache cache) {
28          this.instances = instances;
29          this.selectors = selectors;
30          this.cache = cache;
31      }
32  
33      public T selected(Contextual<T> contextual) {
34          return instances.get(cache.compute(contextual, selectors.keySet(), this::isSelected,
35                  bean -> cacheDurationInMillis(contextual, bean)));
36      }
37  
38      private Selection isSelected(Bean<?> bean) {
39          ContextualSelector selector = selectors.get(bean);
40          try {
41              if (selector == null) {
42                  return Selection.REMAINING;
43              }
44              if (selector instanceof Selector s) {
45                  return Selection.of(s.selected());
46              }
47              return Selection.of(selector.selected(new Context() {
48                  @Override
49                  public Bean<?> bean() {
50                      return bean;
51                  }
52  
53                  @Override
54                  public Object instance() {
55                      return instances.get(bean);
56                  }
57              }));
58          } catch (RuntimeException e) {
59              return Selection.NOT_SELECTED;
60          }
61      }
62  
63      private long cacheDurationInMillis(Contextual<?> contextual, Bean<?> bean) {
64          Feature feature = feature(bean).orElseThrow();
65          long durationMillis = feature.cache().durationMillis();
66          String property = feature.cache().durationMillisProperty();
67          OptionalLong propertyValue = OptionalLong.empty();
68          if (isDefined(property)) {
69              if (property.contains("${type}")) {
70                  String type = "<undefined>";
71                  if (contextual instanceof BeanAttributes) {
72                      type = ((BeanAttributes<?>) contextual).getTypes().stream().filter(t -> !Object.class.equals(t))
73                              .findAny().map(Type::getTypeName).orElse(type);
74                  }
75                  property = property.replace("${type}", type);
76              }
77              propertyValue = ProperyResolver.lookup(property);
78          }
79          return propertyValue.orElse(durationMillis);
80      }
81  
82      private static class ProperyResolver {
83          public static OptionalLong lookup(String property) {
84              return ConfigProvider.getConfig().getOptionalValue(property, Long.class).map(OptionalLong::of)
85                      .orElse(OptionalLong.empty());
86          }
87      }
