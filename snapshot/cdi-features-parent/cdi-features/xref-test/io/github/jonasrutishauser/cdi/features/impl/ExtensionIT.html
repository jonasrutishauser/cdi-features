55      // Workaround as the Weld implementation ignores all build compatible extensions with a SkipIfPortableExtensionPresent annotation.
56      private static Extension forceAddExtension(LiteExtensionTranslator translator,
57              Class<? extends BuildCompatibleExtension> extensionClass) {
58          try {
59              Field utilField = LiteExtensionTranslator.class.getDeclaredField("util");
60              utilField.setAccessible(true);
61              Object util = utilField.get(translator);
62              Field classesField = util.getClass().getDeclaredField("extensionClasses");
63              classesField.setAccessible(true);
64              @SuppressWarnings("unchecked")
65              Map<String, Class<?>> extensionClasses = (Map<String, Class<?>>) classesField.get(util);
66              Field instancesField = util.getClass().getDeclaredField("extensionClassInstances");
67              instancesField.setAccessible(true);
68              @SuppressWarnings("unchecked")
69              Map<Class<?>, Object> extensionClassInstances = (Map<Class<?>, Object>) instancesField.get(util);
70              extensionClasses.put(extensionClass.getName(), extensionClass);
71              extensionClassInstances.put(extensionClass, extensionClass.getDeclaredConstructor().newInstance());
72          } catch (ReflectiveOperationException e) {
73              throw new IllegalStateException(e);
74          }
75          return translator;
76      }

99              String expectedMessage) {

103             initializer.addExtensions(new ConfigExtension());

107 
108         RuntimeException exception = assertThrows(exceptionType, initializer::initialize);

121     @ParameterizedTest(name = "{0}")
122     @MethodSource("cachePropertyvalidation")
123     void validateCacheProperty(Extension extension, Class<? extends RuntimeException> exceptionType,
124             Class<?> beanClass) {
125         SeContainerInitializer initializer = SeContainerInitializer.newInstance().disableDiscovery()
126                 .addExtensions(extension).addBeanClasses(beanClass);
127         boolean shouldFail = false;
128         try {
129             initializer.addExtensions(new ConfigExtension());
130         } catch (NoClassDefFoundError e) {
131             shouldFail = true;
132         }
133 
134         if (shouldFail && extension instanceof FeaturesExtension) {
135             RuntimeException exception = assertThrows(exceptionType, initializer::initialize);
136 
137             assertThat(exception).hasMessageContaining("cache durationMillisProperty must not be set");
138         } else {
139             try (SeContainer container = initializer.initialize()) {
140                 Bean<CacheWithPropertyOnClass> bean = container.select(CacheWithPropertyOnClass.class,
141                         CacheWithPropertyOnClass.class.getAnnotation(Feature.class)).getHandle().getBean();
142                 String expectedProperty = shouldFail ? "" : "some.property";
143                 bean.getQualifiers().forEach(qualifier -> {
144                     if (qualifier instanceof Feature f) {
145                         assertEquals(expectedProperty, f.cache().durationMillisProperty());
146                     }
147                 });
148             }
149         }
